<script>

/* =========================================
   CONFIG
========================================= */

const IMGBB_API_KEY = "YOUR_IMGBB_API_KEY";

/* =========================================
   STATE
========================================= */

let blocks = [];

let activeBlockId = null;

/* =========================================
   HELPERS
========================================= */

function uid() {
    return 'blk_' + Date.now() + '_' + Math.random().toString(36).substring(2,8);
}

function saveStatus(text){
    document.getElementById('saveStatus').innerText = text;
}

function getBlockTitle(block){

    if(block.type === 'text'){
        return block.title || 'Untitled Text';
    }

    if(block.type === 'image'){
        return block.caption || 'Image';
    }

    if(block.type === 'video'){
        return block.caption || 'Video';
    }

    return 'Block';
}

/* =========================================
   ADD BLOCKS
========================================= */

function addTextBlock(){

    blocks.push({
        id: uid(),
        type: 'text',
        title: '',
        subheading: '',
        paragraph: '',
        quote: ''
    });

    refreshUI();
}

function addImageBlock(){

    blocks.push({
        id: uid(),
        type: 'image',
        imageUrl: '',
        caption: ''
    });

    refreshUI();
}

function addVideoBlock(){

    blocks.push({
        id: uid(),
        type: 'video',
        videoUrl: '',
        caption: ''
    });

    refreshUI();
}

/* =========================================
   DELETE
========================================= */

function deleteBlock(id){

    blocks = blocks.filter(b => b.id !== id);

    refreshUI();
}

/* =========================================
   SELECT
========================================= */

function selectBlock(id){

    activeBlockId = id;

    refreshUI();

    const el = document.getElementById(id);

    if(el){
        el.scrollIntoView({
            behavior:'smooth',
            block:'center'
        });
    }
}

/* =========================================
   TREE
========================================= */

function renderTree(){

    const tree = document.getElementById('blockTree');

    tree.innerHTML = '';

    blocks.forEach(block => {

        const div = document.createElement('div');

        div.className = 'treeItem';

        if(block.id === activeBlockId){
            div.classList.add('active');
        }

        div.dataset.id = block.id;

        div.innerHTML = `
            <div class="treeTitle">
                ${getBlockTitle(block)}
            </div>

            <div class="treeMeta">
                ${block.type.toUpperCase()}
            </div>
        `;

        div.onclick = () => selectBlock(block.id);

        tree.appendChild(div);

    });

    new Sortable(tree, {

        animation:150,

        onEnd:(evt)=>{

            const moved = blocks.splice(evt.oldIndex,1)[0];

            blocks.splice(evt.newIndex,0,moved);

            refreshUI(false);

        }

    });

}

/* =========================================
   EDITOR
========================================= */

function renderEditor(){

    const editor = document.getElementById('editor');

    editor.innerHTML = '';

    blocks.forEach(block => {

        const wrap = document.createElement('div');

        wrap.className = 'editorBlock';

        if(block.id === activeBlockId){
            wrap.classList.add('active');
        }

        wrap.id = block.id;

        let html = `
            <div class="blockTop">

                <div class="blockLabel">
                    ${block.type} block
                </div>

                <button class="deleteBtn"
                        onclick="deleteBlock('${block.id}')">
                    Delete
                </button>

            </div>
        `;

        /* ==============================
           TEXT
        ============================== */

        if(block.type === 'text'){

            html += `

                <input
                    type="text"
                    placeholder="Title"
                    value="${block.title || ''}"
                    onchange="updateText('${block.id}','title',this.value)"
                />

                <input
                    type="text"
                    placeholder="Subheading"
                    value="${block.subheading || ''}"
                    onchange="updateText('${block.id}','subheading',this.value)"
                />

                <textarea
                    placeholder="Paragraph"
                    onchange="updateText('${block.id}','paragraph',this.value)"
                >${block.paragraph || ''}</textarea>

                <textarea
                    placeholder="Quote"
                    onchange="updateText('${block.id}','quote',this.value)"
                >${block.quote || ''}</textarea>

            `;
        }

        /* ==============================
           IMAGE
        ============================== */

  if(block.type === 'image'){

    html += `
        <div class="uploadRow">
            <input
                type="text"
                id="imgUrl_${block.id}"
                placeholder="Image URL"
                value="${block.imageUrl || ''}"
                oninput="updateText('${block.id}','imageUrl',this.value)"
            />
            <button class="uploadBtn"
                onclick="setUrl('${block.id}','imageUrl','imgUrl_${block.id}')">
                Set
            </button>
            <input type="file" accept="image/*"
                onchange="uploadImage(event,'${block.id}')" />
        </div>

        <input
            type="text"
            placeholder="Caption"
            value="${block.caption || ''}"
            oninput="updateText('${block.id}','caption',this.value)"
        />
    `;

    if(block.imageUrl){
        html += `<img class="mediaPreview" src="${block.imageUrl}" />`;
    }
}

        /* ==============================
           VIDEO
        ============================== */

  if(block.type === 'video'){

    html += `
        <div class="uploadRow">
            <input
                type="text"
                id="vidUrl_${block.id}"
                placeholder="Video URL"
                value="${block.videoUrl || ''}"
                oninput="updateText('${block.id}','videoUrl',this.value)"
            />
            <button class="uploadBtn"
                onclick="setUrl('${block.id}','videoUrl','vidUrl_${block.id}')">
                Set
            </button>
        </div>

        <div class="smallText">
            Recommended: YouTube / Vimeo / Bunny Stream
        </div>

        <input
            type="text"
            placeholder="Caption"
            value="${block.caption || ''}"
            oninput="updateText('${block.id}','caption',this.value)"
        />
    `;
}

function setUrl(blockId, key, inputId) {
    const input = document.getElementById(inputId);
    if (!input) return;
    updateText(blockId, key, input.value);
    refreshUI();
}

/* =========================================
   UPDATE
========================================= */

function updateText(id,key,value){

    const block = blocks.find(b => b.id === id);

    if(!block) return;

    block[key] = value;

    renderTree();
}

/* =========================================
   IMG UPLOAD
========================================= */

async function uploadImage(event, blockId){

    const file = event.target.files[0];

    if(!file) return;

    saveStatus('Uploading image...');

    try{

        const formData = new FormData();

        formData.append('image', file);

        const response = await fetch(
            `https://api.imgbb.com/1/upload?key=${IMGBB_API_KEY}`,
            {
                method:'POST',
                body:formData
            }
        );

        const data = await response.json();

        if(data.success){

            const block = blocks.find(b => b.id === blockId);

            if(block){

                block.imageUrl = data.data.url;

                refreshUI();

                saveStatus('Image uploaded');

            }

        }

    }catch(err){

        console.error(err);

        saveStatus('Upload failed');

    }

}

/* =========================================
   SAVE
========================================= */

document.getElementById('saveDraftBtn')
.addEventListener('click', ()=>{

    saveStatus('Saving...');

    window.parent.postMessage({

        type:'saveDraft',

        blocks

    }, '*');

    setTimeout(()=>{
        saveStatus('Draft saved');
    },500);

});


/* =========================================
   REFRESH
========================================= */

function refreshUI(scroll=true){

    renderTree();

    renderEditor();

    if(scroll && blocks.length > 0){

        activeBlockId = blocks[blocks.length - 1].id;
    }

}

/* =========================================
   INITIAL
========================================= */

refreshUI();
  
  window.addEventListener('message', (event) => {
    const data = event.data;
    if (!data || !data.type) return;
    if (data.type === 'loadDraft') {
        blocks = data.blocks || [];
        refreshUI();
    }
    // Add this:
    if (data.type === 'requestSave') {
         document.querySelectorAll('input, textarea').forEach(el => el.blur());
    window.parent.postMessage({ type: 'saveDraft', blocks }, '*');
    }
});
